# iOS端集成

在iOS端的podfile中集成。

pod 'WeexAceChart'

# Android集成

在Android端集成

compile 'org.weex.plugin.weexacechart:wx_acechart:0.0.3@aar'
compile 'com.alibaba.dt:acechart:1.1.2.30@aar'

# 前端集成

# Weex支持

在Weex工程中执行下面命令

npm install --save weex-acechart-js

注： 在最新版本的weex-acechart-js中采用了源码发布，方便与工程的weex环境配置兼容。因此需要对webpack配置进行更新，确保babel-loader不会把本库排除。 e.g. 比如使用正则/node_modules(?!(/|\).(weex).)/.

module: {
    // webpack 2.0
    rules: [
      {
        test: /\.js$/,
        use: [{
          loader: 'babel-loader'
        }],
        exclude: /node_modules(?!(\/|\\).*(weex).*)/
      }
}

在vue中，如下方式引入即可使用。

<template>
<weex-ace-chart @onHighlightChanged="onHighlightChanged" :option="option" class="chart"></weex-ace-chart>
</template>


<style scoped>
.chart {
  height: 400px;
  width: 750px;
}
</style>


<script>
import WeexAceChart from "weex-acechart-js";


export default {
  components: { WeexAceChart },
  methods: {
    onHighlightChanged: function(selection) {
      console.log(selection);
    }
  },
  computed: {
    option: function() {
      return {
        "legends": {},
        "xAxises": [],
        "yAxises": [],
        "series": [{
          "displayLegend": true,
          "type": "line",
          "color": "#f1c40f",
          "name": "Line Stroke",
          "points": [
            [1, 2, 3, 4, 5, 6, 7]
          ],
          "lineWidth": 7
        }]
      };
    }
  }
}


</script>
