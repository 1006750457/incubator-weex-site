# FileUploader

文件上传模块

# 方法

# uploadFile(params, successCallback, failedCallback, progressCallback)

上传指定文件

# 参数

params.url {String} 服务器URL。 params.name {String} 文件对于的key，开发者在服务器端通过这个 key 可以获取到文件二进制内容。 params.mime {mime类型} 上传的文件的mime类型。 params.filePath {String} 本地文件路径。 params.header {Object} http请求的header。 params.formData {Object} http请求中其他额外的form data。 sucessCallback {Function} 成功回调，回调函数的参数 ret 的格式如下： ret.data {Object} 服务器返回的数据； ret.statusCode {String} 服务器返回的状态码。 failedCallback {Function} 成功回调，回调函数的参数 ret 的格式如下： ret.data {String} 错误原因； ret.statusCode {String} 状态码，如果有的话。 ret.progressCallback {Function} 进度回调，回调函数的参数 ret 的格式如下： ret.progress 进度(范围是[0,1])； ret.totalBytesSent 已经上传的数据长度，单位 Bytes； ret.totalBytesExpectedToSend 预期需要上传的数据总长度，单位 Bytes。

# 返回值

tag 任务id。可用于cancel。

# cancel(tag)

取消上传文件

# 参数

tag {String} 任务id。

# 示例

<template>
    <div class="wrapper">
        <div class="item-wrap">
            <div class="item camera-photo" @click="cameraPhoto"><text>camera | photo</text></div>
            <div class="item item-last camera-video" @click="cameraVideo"><text>camera | video</text></div>
        </div>
        <div class="item-wrap">
            <div class="item libs-photo" @click="libsPhoto"><text>libs | photo</text></div>
            <div class="item item-last libs-video" @click="libsVideo"><text>libs | video</text></div>
        </div>
        <div class="progress"><Text>进度：{{progress}}</Text></div>
    </div>
</template>

<script>
	const ImagePicker = weex.requireModule('image-picker');
	const fileUploader = weex.requireModule('file-uploader');
    const modal = weex.requireModule('modal');
	export default {
		data() {
			return {
                image: '',
                tag: '',
                progress:0
            }
        },
		methods: {
			_uploadAction() {
                this.tag = fileUploader.uploadFile(
                    {
                        url: 'http://30.8.72.56:3000/file_upload',
                        filePath: this.image,
                        name: 'image',
                        mime: 'image/jpeg',
                        header:{
                            xxx:'yyy'
                        },
                        formData:{
                            aaa:'bbb',
                            ccc:'ddd'
                        }
                    },
                    function successCallback(success) {
                        modal.alert({
                            message: `【Success】${JSON.stringify(success)}`,
                            duration: 1
                        });
                    },
                    function failedCallback(error) {
                        modal.alert({
                            message: `【Err】${JSON.stringify(error)}`,
                            duration: 1
                        });
                    },
                    function progressCallback(progress) {
                        this.progress = progress;
                    }
                );
            },
            libsPhoto() {
	            ImagePicker.pick({
		            type: 'library',
		            mediaType: 'photo'
	            },(result)=>{
	            	console.log('result', result, fileUploader.uploadFile);
		            this.image = result.data.uri;
		            this._uploadAction();
	            });
            },
            libsVideo() {
	            ImagePicker.pick({
                    type: 'library',
                    mediaType: 'video'
	            },(result)=>{
	            	console.log('result', result, fileUploader.uploadFile);
		            this.image = result.data.uri;
		            this._uploadAction();
	            });
            },
            cameraPhoto() {
	            ImagePicker.pick({
		            type:'camera',
		            mediaType:'photo'
	            },(result)=>{
	            	console.log('result', result, fileUploader.uploadFile);
		            this.image = result.data.uri;
		            this._uploadAction();
	            });
            },
            cameraVideo() {
	            ImagePicker.pick({
		            type:'camera',
		            mediaType:'video'
	            },(result)=>{
	            	console.log('result', result, fileUploader.uploadFile);
		            this.image = result.data.uri;
		            this._uploadAction();
	            });
            }
		}
	};
</script>
