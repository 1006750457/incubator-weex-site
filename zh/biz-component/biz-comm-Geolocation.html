# Geolocation

用来获取当前设备的经纬库和详细地址。分为一次获取和持续监听。

# 方法

# getCurrentPosition(successCallback, errorCallback, options)

获取当前设备的经纬库和详细信息。

# 参数

options.enableHighAccuracy {Object} 是否获取高精度的位置，true/false，默认值：false； options.address 是否获取地址描述，如中国杭州市，true/false，如果不获取地址描述，则可以更快的返回经纬度，默认为 false； successCallback {Function} 获取定位信息的成功回调，回调函数的参数 ret 的格式如下： ret.coords； ret.latitude； ret.longitude； ret.accuracy； ret.altitude 海拔，负数为低于海平面（仅支持iOS）； ret.altitudeAccuracy 海拔精确度，负值无效（仅支持iOS）； ret.heading 设备行进方向，北方向为0°，顺时针计算，东为90°，负值无效（仅支持iOS）； ret.speed 在行进方向上的设备速度，单位m/s，负值无效（仅支持iOS）。 ret.address.city； ret.address.province； ret.address.area； ret.address.road； ret.address.addressLine； errorCallback {Function} function(ret){} 获取定位信息的失败回调。回调函数的参数 ret 的格式如下： ret.errCode {Number} 失败码，可选值： SUCCESS - 90000； NO_PERMISSION_ERROR - 90001； PARAMS_ERROR - 90002； LOCATION_ERROR - 9003； LOCATION_TIME_OUT - 9004； ret.errMsg {String} 失败信息，可选值： NO_PERMISSION_ERROR - "NO PERMISSION"； PARAMS_ERROR - "PARAMS_ERROR"； LOCATION_ERROR - "LOCATION_FAIL"； LOCATION_TIME_OUT - "LOCATION_TIME_OUT"； SUCCESS - "SUCCESS"。

# watchPosition(successCallback, errorCallback, options)

监听定位信息。

# 参数

同 getCurrentPosition()。

# 返回值

{String} watchID 停止监听时调用。

# clearWatch(watchId)

删除定位监听。

# 参数

watchId {String} 注册监听时返回的唯一标示。

# isLocationEnabled(callback)

判断定位服务是否开启。

# 参数

callback {String} 返回定位服务是否开启

# goToLocationSetting

跳转到定位服务开启设置页

<template>
  <scroller style="flex-direction: column;">
    <text title="TC_BizModule_Geolocation"></text>
    <div>
        <text class= "btn" value = '{{location}}' onclick = "getCurrentPosition"></text>
        <text class= "btn" value = '{{watchlocation}}' onclick = "watchPosition"></text>
        <text class= "btn" value = '{{clearId}}' onclick ="clearWatch"></text>
        <text class= "btn" value = '{{error_args}}' onclick ="getCurrentPositionWithErrorArgs"></text>
    </div>
    <text></text> title="location 响应状态："></text>
    <div style="flex-direction:column; margin-top:5; padding-top:5; padding-bottom:5;">
        <text class="textarea" >{{resultType}}</text>
    </div>
    <text title="location请求结果："></text>
    <div style="flex-direction:column; margin-top:5; padding-top:5; padding-bottom:5;">
        <text class="textarea">{{resultData}}</text>
    </div>
  </scroller>
</template>
<style>
.btn{ width:710; font-size: 35; border-width: 3; border-color: #cccccc; padding: 10; border-style: solid;border-radius: 3; margin: 5;font-family: Times New Roman; }
.textarea{width: 710; height: 400; font-size: 30;}
</style>
<script>
  module.exports = {
    data: {
            location: "getCurrentPosition",
            watchlocation: "watchlocation",
            btntxt_clearWatch: "clear id",
            resultType:"type",
            resultData:"resultData",
            clearId:"clearId",
            error_args:"error_args_getCurrentPosition",
            i: 0
    },
    methods: {
      //getCurrentPosition
      getCurrentPosition:function (e) {
        var geolocation = require('@weex-module/geolocation');
            // success
            geolocation.getCurrentPosition(function (result) {
              console.log("into--[getCurrentPosition success] result:" + JSON.stringify(result));
                // this.location = JSON.stringify(result);
                this.resultData = JSON.stringify(result);
                this.resultType = "success";
            }.bind(this),
            // 失败
            function (result) {
                console.log("into--[getCurrentPosition error] result:" + JSON.stringify(result));
                  this.resultData = JSON.stringify(result);
                  this.resultType = "Error";
            }.bind(this),
            // 参数
            {enableHighAcuracy: false, address: true});
        },
        //getCurrentPositionWithErrorArgs
      getCurrentPositionWithErrorArgs:function (e) {
        var geolocation = require('@weex-module/geolocation');
            // success
            geolocation.getCurrentPosition(function (result) {
              console.log("into--[getCurrentPositionWithErrorArgs success] result:" + JSON.stringify(result));
                // this.location = JSON.stringify(result);
                this.resultData = JSON.stringify(result);
                this.resultType = "success";
            }.bind(this),
            // 失败
            function (result) {
                console.log("into--[getCurrentPositionWithErrorArgs error] result:" + JSON.stringify(result));
                  this.resultData = JSON.stringify(result);
                  this.resultType = "Error";
            }.bind(this),
            // 参数
            {
              error:"error"
            });
        },
      // watchPosition
      watchPosition:function(e){
        var geolocation = require('@weex-module/geolocation');
        // 成功
        geolocation.watchPosition(function (result) {
                console.log("into--[watchPosition success] result:" + JSON.stringify(result));
                var str = this.watchlocation + '\n' + JSON.stringify(result);
                this.resultData = str;
                this.resultType ="watchPosition success ";
                this.clearId = result.watchId ;
                this.i += 1;
                if (this.i == 2) {
                    console.log("into--position update to:  " + i );
                    geolocation.clearWatch(result.watchId);
                    this.btntxt_clearWatch = result.watchId + " clean";
                    console.log(result.watchId + " has been cleaned " );
                    require('@weex-module/modal').toast({
                        'message': "[clear watch] id:"+result.watchId,
                        'duration': 1
                    });
                }
            }.bind(this),
            // 失败
            function (result) {
                console.log("into--[watchPosition success] result:" + JSON.stringify(result));
                // this.watchlocation = JSON.stringify(result);
                this.resultData = JSON.stringify(result);
                this.resultType ="watchPosition Error "
            }.bind(this),
            // 参数
            {enableHighAcuracy: false, address: true});
      },
      // clearWatch
      clearWatch:function(e){
        var geolocation = require('@weex-module/geolocation');
        geolocation.clearWatch(this.watchId);
        this.resultType = this.watchId;
        this.resultData = this.clearId + " cleaned.."
      },
    }
  }
</script>
