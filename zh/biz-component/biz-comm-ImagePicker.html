# ImagePicker

图片/视频拍摄、选取。

# 方法

# pick(options, callback)

图片/视频拍摄、选取。

# 参数

options.type {String} 选取类型，可选值：图库(library)或者相机(camera)； options.mediaType {String} 媒体类型，可选值：相片(photo)或者视频(video)； options.quality {Number} 图片质量，值范围为[0,1](仅iOS 有效)； options.videoQuality {String} 视频质量，可选值： iOS支持 low／medium／high 三种； android支持 low／high； durationLimit {Number} 视频时长，单位：秒。 callback {Function} function(ret){} 用户操作完成后的回调，回调函数的参数 ret 的格式如下： ret.data.uri {String} 文件地址； ret.data.width {String} 宽(仅iOS 有效)； ret.data.height {String} 高(仅iOS 有效)； ret.data.fileSize {String} 文件尺寸，单位：kb。

  <div>
    
    
      <div style="flex-direction:row">
        <div style="justify-content:center;align-items:center">
          <text @click="func1" style="width:375;height:200;background-color:#00ff00;justify-content:center;align-items:center">camear photo</text>
        </div>
        <div style="justify-content:center;align-items:center">
          <text @click="func2" style="width:375;height:200;background-color:#ff0000;justify-content:center;align-items:center">camear video</text>
        </div>
      </div>
      
       <div style="flex-direction:row">
         <div style="justify-content:center;align-items:center">
           <text @click="func3" style="width:375;height:200;background-color:#0000ff;justify-content:center;align-items:center">libary photo</text>
           </div>
           
             <div style="justify-content:center;align-items:center">
               <text @click="func4" style="width:375;height:200;background-color:#ff00ff;justify-content:center;align-items:center">libary video</text>
           </div>
        
        
      </div>
    
      <image :src="image" style="width:750;height:300;background-color:#e0e0e0"></image>
         
       <video class="video" :src="src" autoplay controls
    @start="onstart" @pause="onpause" @finish="onfinish" @fail="onfail"></video>

 
  </div>
</template>

<script>

var ImagePicker = weex.requireModule('image-picker')
  export default {
    data () {
      return {
        state: '----',
        src:'',
        image:''
      }
    },
    methods:{
      func1(){
  		  ImagePicker.pick({
  		    type:'camera',
  		    mediaType:'photo'
  		  },(result)=>{
  		    console.log('result->',JSON.stringify(result));
  		    this.image = result.data.uri;
  		  });
        
      },
     func2(){
  		  ImagePicker.pick({
  		    type:'camera',
  		    mediaType:'video'
  		  },(result)=>{
  		    console.log('result->',JSON.stringify(result));
  		    this.src = result.data.uri;
  		  });
      },
      func3(){
  		  ImagePicker.pick({
  		    type:'library',
  		    mediaType:'photo'
  		  },(result)=>{
  		    console.log('result->',JSON.stringify(result));
  		    this.image = result.data.uri;
  		  });
      },
      func4(){
  		  ImagePicker.pick({
  		    type:'library',
  		    mediaType:'video'
  		  },(result)=>{
  		    console.log('result->',JSON.stringify(result));
  		    this.src = result.data.uri;
  		  });
      },
      onstart (event) {
        this.state = 'onstart'
      },
      onpause (event) {
        this.state = 'onpause'
      },
      onfinish (event) {
        this.state = 'onfinish'
      },
      onfail (event) {
        this.state = 'onfinish'
      }
    }
  }
</script>
